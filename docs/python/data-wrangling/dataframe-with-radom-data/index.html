    import numpy as np
import pandas as pd
import random
    def xrange(x):
    return iter(range(x))
    rnd_1 = [random.randrange(1, 20) for x in xrange(10)]
    rnd_1
    [16, 3, 18, 19, 1, 13, 17, 3, 4, 3]

    rnd_2 = [random.randrange(1, 20) for x in xrange(10)]
    rnd_2
    [12, 9, 14, 5, 16, 14, 11, 11, 12, 10]

    date = pd.date_range('2018-01-01', '2018-01-10')
    data = pd.DataFrame({
    'date' : date,
    'rnd_1' : rnd_1,
    'rnd_2' : rnd_2
})
    data
