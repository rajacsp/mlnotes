    class City(Base):
     __tablename__ = 'city'

     id = Column(Integer, primary_key=True)
     name = Column(String)
     province = Column(String)
     country = Column(String)

     def __repr__(self):
        return "<City(name='%s', province='%s', country='%s')>" % (
                             self.name, self.province, self.country)
