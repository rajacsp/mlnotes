    Match All String

    20 Apr 2019 
    one = "Toronto is nice"
    one
    'Toronto is nice'

    one_array = one.split(" ")
    two = "It is nice to be in Toronto"
    two_array = two.split(" ")
    two_array
    ['It', 'is', 'nice', 'to', 'be', 'in', 'Toronto']

    all(x in two for x in one_array)
    True
