import re
from nltk.tokenize import word_tokenize
from collections import Counter

word_counter = Counter(word_tokenize("""The cat is in the box. The cat likes the box. The box is over the cat."""))

word_counter

         'cat': 3,
         'is': 2,
         'in': 1,
         'the': 3,
         'box': 3,
         '.': 3,
         'likes': 1,
         'over': 1})

word_counter.most_common(4)
