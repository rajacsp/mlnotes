title: "Chainmap" author: "Rj" date: 2019-04-20 description: "List Test" type: technical_note draft: false

city = {'name': 'Toronto', 'short_name': 'TO'}

number = {'a': 2, 'b': 3, 'c' : 4}

city

{'name': 'Toronto', 'short_name': 'TO'}

number

{'a': 2, 'b': 3, 'c': 4}

from collections import ChainMap

ab = ChainMap(city, number)

ab

ChainMap({'name': 'Toronto', 'short_name': 'TO'}, {'a': 2, 'b': 3, 'c': 4})

type(ab)

collections.ChainMap

for k, v in ab.items():
    print(k , "==>", v)

short_name ==> TO
b ==> 3
c ==> 4
a ==> 2
name ==> Toronto

ab_list = list(ChainMap(city, number))

ab_list

['short_name', 'b', 'c', 'a', 'name']

ab_tuple = tuple(ChainMap(city, number))

ab_tuple

('short_name', 'b', 'c', 'a', 'name')
