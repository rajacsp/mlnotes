import requests
import pandas as pd

filename = 'https://gitlab.com/rajacsp/datasets/raw/master/amazon_co_ecommerce_sample.csv'

r = requests.get(filename)
data = r.content

df = pd.read_csv(BytesIO(data), index_col=0)
