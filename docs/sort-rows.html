import pandas as pd

data = {
    'language' : [80, 67, 90],
    'maths' : [100, 78, 98],
    'science' : [98, 77, 56],
    'programming': [100, 100, 90]
}

index = ['chris', 'kevin', 'peter']

df = pd.DataFrame(data, index = index)

df

peter 90 98 56 90
