import tensorflow as tf

import os

# Just disables the warning, doesn't enable AVX/FMA

tf.InteractiveSession()

a = tf.zeros((2, 2))
b = tf.ones((2, 2))

abc = tf.reduce_sum(b, reduction_indices=1).eval()
#rray([2., 2.], dtype=float32)
print(abc)

print(a.get_shape())

# reshape
ef = tf.reshape(a, (1, 4)).eval()
print(ef)
