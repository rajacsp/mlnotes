def fib(n):
    if n == 0:
        return 0
    elif n == 1:
        return 1
    else:
        return fib(n-1) + fib(n-2)

print(fib(8))

def fibonacci(n):
    a = b = 1

    for i in range(n):
        yield a
        a, b = b, a+b

print(2)
