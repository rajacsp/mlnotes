import time
import random as rd

team_1_score = 0
team_2_score = 0

def play_over(over, t_score):

    print('Over ', over)

    for x in range(6):

        score = rd.randint(0, 6)
        t_score += score
        print('Ball ', (x+1), score)

        time.sleep(0.5)

    print('')
    return t_score

# Team 1 Batting

team_1_score = play_over(1, team_1_score)
team_1_score = play_over(2, team_1_score)
