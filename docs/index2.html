from nltk.stem.cistem import Cistem

c_stemmer = Cistem()

c_stemmer.stem("filtering")

'filtering'

c_stemmer.segment("filtering")

('filtering', '')

c_stemmer.segment("Ausgefeiltere")

('ausgefeilt', 'ere')

segment method will return both the stem and the rest that was removed at the end
