# print all cols

new_cols = {}
for old_col in list(df.columns):
    # print(col)

    col = old_col
    col = col.lower()

    col = col.replace(".", "")
    col = col.replace(" ", "_")
    col = col.replace("(", "")
    col = col.replace(")", "")
    col = col.replace("%", "")
    col = col.replace("/", "")
    col = col.replace("$", "")

    # if " " in col or "(" in col:
    # print(col)

    new_cols[old_col] = col
